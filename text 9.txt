-- Texto do loadstring que ser√° copiado
local gravityLoadstring = [[loadstring(game:HttpGet("https://raw.githubusercontent.com/Dev-GravityHub/BloxFruit/refs/heads/main/Main.lua"))()]]

-- Bot√£o pra copiar link / c√≥digo do Gravity Hub

-- Caixa escondida pra mostrar o c√≥digo se setclipboard falhar
local textbox = Instance.new("TextBox")
textbox.Size = UDim2.new(0.9, 0, 0.28, 0)
textbox.Position = UDim2.new(0.05, 0, 0.33, 0)
textbox.BackgroundColor3 = Color3.fromRGB(30,30,30)
textbox.TextColor3 = Color3.fromRGB(255,255,255)
textbox.TextWrapped = true
textbox.ClearTextOnFocus = false
textbox.TextEditable = true
textbox.Text = gravityLoadstring
textbox.Visible = false
textbox.Font = Enum.Font.Code
textbox.TextScaled = false
textbox.Parent = frame

local tbCorner = Instance.new("UICorner", textbox)
tbCorner.CornerRadius = UDim.new(0,6)

copy.MouseButton1Click:Connect(function()
	-- tenta copiar pro clipboard primeiro
	local ok, err = pcall(function()
		setclipboard(gravityLoadstring)
	end)
	if ok then
		status.Text = "C√≥digo copiado!"
		-- some daqui 2s
		task.delay(2, function() status.Text = "" end)
	else
		-- se n√£o der pra copiar, mostra o TextBox com o c√≥digo pra copiar manual
		status.Text = "Clipboard indispon√≠vel ‚Äî copie manualmente."
		textbox.Visible = true
		-- seleciona todo o texto (alguns executores focam automaticamente)
		pcall(function() textbox:CaptureFocus() end)
		-- limpa msg depois de um tempo
		task.delay(4, function()
			status.Text = ""
		end)
	end
end)

-- Efeito de aparecer suave
frame.BackgroundTransparency = 1

subtitle.TextTransparency = 1
copy.TextTransparency = 1
close.TextTransparency = 1
textbox.TextTransparency = 0 -- textbox visible text should stay readable

for i = 1, 10 do
	task.wait(0.05)
	frame.BackgroundTransparency = 1 - (i * 0.09)
	title.TextTransparency = 1 - (i * 0.09)
	subtitle.TextTransparency = 1 - (i * 0.09)
	copy.TextTransparency = 1 - (i * 0.09)
	close.TextTransparency = 1 - (i * 0.09)
end
